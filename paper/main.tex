\subsection{Data}

Let $\mathrm{L}\in\N_{\geq 1}$ be a grid extent,
and $\mathrm{U}\in\N_{\geq 1}$ be an upsampling factor,
and $\mathrm{S}\in\N_{\geq 1}$ be a number of snakes.

\begin{definition}[Grid]
$$
	\mathcal{G}
	=
	[1..\mathrm{L}]^{2}
$$
\end{definition}

\begin{definition}[Action Space]
$$
	\mathcal{A}
	=
	\{(0,0),(0,1),(0,-1),(-1,0),(1,0)\}
$$
\end{definition}

\begin{definition}[Movement Space]
$$
	\mathcal{M}
	=
	\mathcal{A}\setminus\{(0,0)\}
$$
\end{definition}

\begin{definition}[Snake Space]
$$
	\mathcal{S}
	=
	\bigcup_{n\geq 2}\mleft\{
		\bm{s}\in(\Z\times\Z)^{n}
		\setmid
		\forall i\in[1..n-1]\mleft(
			s_{i+1}-s_{i}\in\mathcal{M}
		\mright)
	\mright\}
$$
\end{definition}

\begin{definition}[Food Space]
$$
	\mathcal{F}
	=
	\mathcal{G}\cup\{\perp\}
$$
\end{definition}

\begin{definition}[Action Oracle]
$$
	\delta:\mathcal{S}\times\N\to\mathcal{A}
$$
\end{definition}

\begin{definition}[Snake Occupancy]
Let $\bm{S}\in\mathcal{S}^{*}$.
$$
	\mathcal{O}(\bm{S})
	=
	\mleft\{
		s_{i}
		\mid
		\bm{s}\in\bm{S}
		\land
		i\in[1..\abs{\bm{s}}]
	\mright\}
$$
\end{definition}

\begin{definition}[Pairing Function]
\begin{align*}
	\rho
	&:
	\mathcal{G}^{2\mathrm{S}}
	\to
	(\mathcal{G}\times\mathcal{G})^{\mathrm{S}}
	\\
	\rho(\bm{x})
	&=
	(
		(x_{2i-1},x_{2i})
	)_{i=1}^{\mathrm{S}}
\end{align*}
\end{definition}

\begin{algorithm}[H]
\caption{$\textsc{Game}$}
\begin{algorithmic}
	\State ${
		\bm{S}
		\sim
		\rho\circ\mathcal{U}(
			\{
				\bm{s}\in\mathcal{G}^{2\mathrm{S}}
				\mid
				\forall i\in[1..\mathrm{S}](
					s_{2i}-s_{2i-1}\in\mathcal{M}
				)
				\land
				|\{s_{i}\}_{i=1}^{2\mathrm{S}}|
				=
				2\mathrm{S}
			\}
		)
	}$

	\State $E\gets\mathcal{G}\setminus\mathcal{O}(\bm{S})$
	\If{$E=\varnothing$}
		\State $f\gets\perp$
	\Else{}
		\State $f\gets\mathcal{U}(E)$
	\EndIf

	\State $t\gets 0$

	\While{$|\bm{S}|>0$}
		\State $i_{f}\gets 0$

		\ForAll{$\bm{s}\bm{\in}\bm{S}$}
			\Comment{Translate snakes}
			\State $d\gets\delta(\bm{s},t)$
			\If{$d=(0,0)$}
				\State $h\gets 2s_{1}-s_{2}$
			\Else{}
				\State $h\gets s_{1}+d$
			\EndIf
			\State $\bm{s}\gets h\concat(s_{i})_{i=1}^{|\bm{s}|-1}$
		\EndFor

		\ForAll{$\bm{s}\bm{\in}\bm{S}$}
		\Comment{Grow snakes}
			\If{$s_{1}=f$}
				\State ${
					\bm{s}
					\gets
					\bm{s}
					\concat
					(2s_{|\bm{s}|}-s_{|\bm{s}|-1})
				}$
				\State $i_{f}\gets 1$
			\EndIf
		\EndFor

		\State $D\gets\varnothing$
		\State ${
			B
			\gets
			\{
				s_{i}
				\mid
				\bm{s}\bm{\in}\bm{S}
				\land
				i\in[2..|\bm{s}|]
			\}
		}$
		\ForAll{$\bm{s}\bm{\in}\bm{S}$}
			\If{$s_{1}\not\in\mathcal{G}\lor s_{1}\in B$}
				\State $D\gets D\cup\{\bm{s}\}$
			\EndIf
		\EndFor

		\State ${
			H
			\gets
			\{
				\bm{g}\in\mathcal{G}
				\mid
				\exists\bm{c}\bm{\subseteq}\bm{S}(
					|\bm{c}|\geq 2
					\land
					\forall\bm{s}\bm{\in}\bm{c}(s_{1}=\bm{g})
				)
			\}
		}$
		\ForAll{$h\in H$}
			\State $C\gets\{\bm{s}\bm{\in}\bm{S}\mid s_{1}=h\}$
			\State ${
				w
				\gets
				\mathcal{U}(
					\arg\max_{\bm{s}\in C}
					|\bm{s}|
				)
			}$
			\State $D\gets D\cup (C\setminus\{w\})$
		\EndFor

		\ForAll{$d\in D$}
		\Comment{Kill snakes}
			\State $\bm{S}\gets\bm{S}\bm{\setminus}\{d\}$
		\EndFor

		\If{$i_{f}=1$}
		\Comment{Respawn food}
			\State $E\gets\mathcal{G}\setminus\mathcal{O}(\bm{S})$
			\If{$E=\varnothing$}
				\State $f\gets\perp$
			\Else{}
				\State $f\gets\mathcal{U}(E)$
			\EndIf
		\EndIf

		\State $t\gets t+1$
	\EndWhile
\end{algorithmic}
\end{algorithm}

\subsection{Architecture}

Let $\mathrm{R}\in\N_{\geq 1}$ be a representation space dimensionality,
and $\mathrm{Z}\in\N_{\geq 1}$ be a latent space dimensionality,
and $\mathcal{X}=\N_{\leq 255}^{\mathrm{U}\mathrm{L}\times\mathrm{U}\mathrm{L}\times 3}$ be a raw space,
and $\mathcal{R}=\R^{\mathrm{R}}$ be a representation space,
and $\mathcal{Z}=\R^{\mathrm{Z}}$ be a latent space,
and $\mathbb{P}_{\mathcal{Z}}$ be a probability measure on $\mathcal{Z}$,
and $\gamma\in\R_{[0,1]}$ be a discount factor,
and $T\in\N_{\geq 1}$ be a planning horizon.

\begin{definition}[Colour Function]
\begin{align*}
	\mathcal{C}
	&:
	\mathcal{G}
	\times
	\mathcal{S}^{*}
	\times
	\mathcal{F}
	\to
	\N_{\leq 255}^{3}
	\\
	\mathcal{C}((i,j),\bm{S},f)
	&=
	\begin{cases}
		(128,128,255) % Playable head
		, &
		\text{if }
		\abs{\bm{S}}\geq 1\land(i,j)=s_{1,1}
		\\
		(0,0,255) % Blue
		, &
		\text{if }
		\abs{\bm{S}}\geq 1
		\land
		\exists k\in\mleft[2..\abs{\bm{s}_{1}}\mright]\mleft(
			(i,j)=s_{1,k}
		\mright)
		\\
		(255,128,128) % Nonplayable head
		, &
		\text{if }
		\exists r\in\mleft[2..\abs{\bm{S}}\mright]\mleft(
			(i,j)=s_{r,1}
		\mright)
		\\
		(255,0,0) % Red
		, &
		\text{if }
		\exists r\in\mleft[2..\abs{\bm{S}}\mright],
		\exists k\in[2..\abs{\bm{s}_{r}}]\mleft(
			(i,j)=s_{r,k}
		\mright)
		\\
		(0,255,0) % Green
		, &
		\text{if }
		f\neq\perp\land(i,j)=f
		\\
		(0,0,0) % Black
		, &
		\text{otherwise}
	\end{cases}
\end{align*}
\end{definition}

\begin{definition}[Display Function]
\begin{align*}
	\mathcal{D}
	&:
	\mathcal{S}^{*}
	\times
	\mathcal{F}
	\to
	\mathcal{X}
	\\
	\mathcal{D}(\bm{S},f)
	&=
	\mleft(
		\mathcal{C}\mleft(
			\mleft(
				\mleft\lceil\frac{i}{\mathrm{U}}\mright\rceil,
				\mleft\lceil\frac{j}{\mathrm{U}}\mright\rceil
			\mright),
			\bm{S},
			f
		\mright)
	\mright)_{i,j\in[1..\mathrm{U}\mathrm{L}]}
\end{align*}
\end{definition}

\begin{definition}[Encoder]
$$
	\phi:\mathcal{X}\to\mathcal{R}
$$
\end{definition}

\begin{definition}[World Model]
$$
	\mathscr{F}
	:
	\mathcal{R}\times\mathcal{A}\times\mathcal{Z}
	\to
	\mathcal{R}
$$
\end{definition}

\begin{definition}[Energy]
$$
	\mathscr{C}:\mathcal{R}\to\R_{\geq 0}
$$
\end{definition}

\begin{definition}[Discounted Sum Planner]
\begin{align*}
	f
	&:
	\mathcal{X}
	\to
	\bigsqcup_{t\geq 1}\mathcal{A}^{t}
	\\
	f(x)
	&=
	\mathcal{U}\mleft(
		\arg\min_{\bm{a}\in\mathcal{A}^{T}}
		\E_{\bm{z}\sim\mathbb{P}_{\mathcal{Z}}^{\otimes T}}\mleft[
			\sum_{t=1}^{T}
			\gamma^{t-1}
			\mathscr{C}\mleft(
				\mleft(
					\bigcirc_{i=1}^{t}
					\mathscr{F}\mleft(\cdot,a_{i},z_{i}\mright)
				\mright)(
					\phi(x)
				)
			\mright)
		\mright]
	\mright)
\end{align*}
\end{definition}

\section{Appendix}

\begin{notation}
Let $x$ be a set, and $\bm{y}$ be a sequence.
$$
	x\bm{\in}\bm{y}
	\equiv
	x\in\im(\bm{y})
$$
\end{notation}

\begin{notation}
Let $\bm{x},\bm{y}$ be sequences.
$$
	\bm{x}\bm{\subseteq}\bm{y}
	\equiv
	\im(\bm{x})\subseteq\im(\bm{y})
$$
\end{notation}

\begin{notation}
Let $\bm{x}$ be a sequence, and $y$ be a set.
$$
	\bm{x}\bm{\setminus}y
	=
	(x_{i})_{i\in\{j\in\dom(\bm{x})\mid x_{j}\not\in y\}}
$$
\end{notation}

\begin{notation}
Let $\bm{x}$ be a sequence.
$$
	\abs{\bm{x}}
	=
	\abs{\dom(\bm{x})}
$$
\end{notation}

\begin{notation}
Let $n\geq 1$, and $X$ be a set, and $f_{1},\dots,f_{n}:X\to X$.
$$
	\bigcirc_{i=1}^{n}f_{i}
	=
	f_{n}\circ\dots\circ f_{1}
$$
\end{notation}
